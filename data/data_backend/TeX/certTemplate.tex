% Format für Zeugnistexte
\newbox\certContent
\newcommand{\certText}[4]{%
  \setbox\certContent = \vbox{
    \ifthenelse{%
      \equal{#1}{}%
    }%
    {}%
    {%
      {\large{\textbf{#1}}}%
      %
      \vspace{0.5\baselineskip}
      %
    }
  
    \ifthenelse{%
      \equal{#2}{}%
    }%
    {}%
    {%
      \nopagebreak\par
      #2
      \nopagebreak
      \vspace{0.5\baselineskip}
    }
    \ifthenelse{\equal{#3}{}}{}{%
      \nopagebreak\par
      \emph{#3}%
      \nopagebreak
      \vspace{0.5\baselineskip}
    }
    \ifthenelse{\equal{#4}{}}{}{\par\hspace{\fill}\emph{#4}}%  
    \hfil\rule{\textwidth}{0.4pt}\hfil%
    \vskip\baselineskip%
  }
  %\ifdim\thisPageTakenSpace>0pt\end{minipage}\fi
  \ifdim\pagetotal>0pt
    \ifdim\dimexpr\pagegoal-\pagetotal+\baselineskip\relax<\ht\certContent
      \newpage%
      \raisebox{\baselineskip}{\hfil\rule{\textwidth}{0.4pt}\hfil}%
    \fi%
  \fi%
  \unvbox\certContent%
  \par%
}
